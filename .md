function scrollToLabel(tl, label) {
  const st = tl.scrollTrigger;
  const time = tl.labels[label];
  const progress = time / tl.duration();
  const y = st.start + progress * (st.end - st.start);
  gsap.to(window, { scrollTo: y, duration: 1 });
}
